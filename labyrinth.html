Project specification

This program utilizes graph traversal algorithm - depth first search to find the path from start to end for a maze
Main implementation:


void Maze::DFS(int start_vertex, int end_vertex){
/*if end_vertex spotted, stop the recursive call */
if(visited[end_vertex] != 1){
 visited[start_vertex] = 1;
 path.push_back(start_vertex);
  for(unsigned int i = 0; i < graph[start_vertex].size(); i++){
  //If adjacent vertices exist and not visited, call DFS recursively
   if(graph[start_vertex][i] == 1 && visited[i] == 0){
    DFS(i, end_vertex);
   }
  }
  //If end_vertex spotted, stop the recursive call
  if(visited[end_vertex] == 1)
   return;
  else{
   visited[start_vertex] = 2;
   path.pop_back();
  }
}
        return;
}
